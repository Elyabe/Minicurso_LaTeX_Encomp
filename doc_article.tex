			Na forma vetorial, fica $$\vec{f} = m \cdot \vec{a}.$$
			
			Num triângulo qualquer, vale a Lei dos Cossenos
			\begin{equation}
				a^2 + b^2 - 2\cos(\theta) = c^2\mbox{, com } \theta = \hat{A}.
				\label{eq:leiCos}
			\end{equation}
			Em particular, num triângulo retângulo, $\theta = \frac{\pi}{2}$.
			Daí
			\begin{math}
				 a^2 + b^2 = c^2
			\end{math}
			
			Algumas identidades trigonométricas...
			
			\begin{eqnarray}
				\sin^2(x) + \cos^2(x) &= &1 \label{eq: relTrigFund}\\ 
				1 + \tan^2(x) &= &\sec^2(x) \\ \nonumber
				\cos(2\theta) &= &\cos^2(\theta) - \sin^2(\theta)
			\end{eqnarray}
		
			Função contínua
			\begin{math}
				\lim\limits_{x \rightarrow a} f(x) = f(a)

			\begin{equation}

		\begin{algorithm}[h]
			\caption{{\sc Merge }}
			\DontPrintSemicolon % Some LaTeX compilers require you to use \dontprintsemicolon instead
			\KwIn{A[], p, q, r : inteiro}
			\KwOut{Sequência A[p, $\ldots$ , r-1] ordenada}
			$i, j, k,$ temp[$1, \ldots, r-p$]: inteiro 
			
			$i \gets p$;\\
			$j \gets q$;\\
			$k \gets 1$;\\
			
			\While{$i < q$ \textbf{e} $j < r$ }
			{
				\label{alg:laco}
				\If{A[$i$] $\leq$ A[$j$] }{temp[$k$] $\gets$ A[$i$];\\ $i \gets i + 1$; }
				\Else{temp[$k$] $\gets$ A[$j$];\\ $j \gets j + 1$; }
				$k \gets k + 1$;
			}
			
			\While{$i < q$ }
			{
				temp[$k$] $\gets$ A[$i$];\;
				$i \gets i + 1$;\;
				$k \gets k + 1$;
			}
			
			\While{$j < r$ }
			{
				temp[$k$] $\gets$ A[$j$];\; 
				$j \gets j + 1$;\;
				$k \gets k + 1$;
			}			
			
			A[$p , \ldots, r-1$] $\gets$ temp[$1, \ldots, r-p$]\;
			\Return{A}\;
			\label{algo:selectionsort}
		\end{algorithm}
